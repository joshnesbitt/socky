h1. Socky

A quick and dirty socket helper library.

h2. Usage

h3. Setup

Socky::Base.setup do |config|
  
  config.host  = "somewhere.com"
  config.port  = 12345
  config.debug = true
  config.rescue_retries = 3
  config.rescue_classes = [ Socky::Errors::Timeout,
                            Socky::Errors::Socket ]
  
end


h3. Normal usage

request = Socky.new
request.data << "some data"
request.make!


h3. Block usage

Socky::Base.send do |data|
  
  data << "some data"
  
end # sends data and end of block
